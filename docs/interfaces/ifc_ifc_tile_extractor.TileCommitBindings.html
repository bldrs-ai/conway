Interface TileCommitBindings

The wasm module surface the extractor commits through: the tile pool bindings plus the Geometry-taking commit (embind commitGeometryTile, which serialises a reified Geometry's payload straight into pool chunks).

interface TileCommitBindings {
    commitGeometryTile(assetID: number, geometry: unknown): boolean;

    geometryTileSegmentCount(assetID: number): number;
    geometryTileVertexByteLength(assetID: number): number;
    initGeometryTilePool(budgetBytes: number, chunkBytes: number): boolean;
    releaseGeometryTile(assetID: number): boolean;
    retainGeometryTile(assetID: number): boolean;
}
